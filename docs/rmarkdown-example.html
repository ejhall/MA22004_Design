x <- seq(-8, 8, length=1000)
y0 <- dnorm(x, -2, 1)
y1 <- dnorm(x, 2, 1)
df <- tibble(x, y0, y1)
df <- melt(df, id.var = "x", value.name = "y")

ggplot(data = df, aes(x = x, color = variable)) + geom_line(aes(y=y))
